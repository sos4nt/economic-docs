def to_api_hash
  hash = {}
  self.class.attributes.each do |name, options|
    converter = Economic.converter_for_type(options[:type])
    value = read_attribute(name)
    if attribute_nillable?(name) || !value.nil?
      hash[name.to_s.camelcase] = converter.to_api(value)
    end
  end
  hash
end

def write_attribute(name, value)
  raise MissingAttributeError, "Missing attribute: '#{name}'." if attribute_missing?(name)
  attributes_before_typecast[name] = value
  typed_value = typed_value_for(name, value)
  attributes[name] = typed_value
end
