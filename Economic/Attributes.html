def api_attribute(name, options={})
  # TODO: use ApiArgument
  hash = {}
  hash[:name] = options.delete(:name)
  hash[:type] = options.delete(:type)
  hash[:nillable] = options.delete(:nillable)
  attributes[name] = hash
  raise "Unknown api_attribute options in #{self.name}: #{options.inspect}" unless options.empty?
  if hash[:type] =~ /(_handle|_data)\z/
    Economic.const_get(hash[:type].to_s.camelcase)
  else
    Economic.const_get("#{hash[:type].to_s.camelcase}Converter")
  end
  create_accessors(name)
end

def attribute_nillable?(name)
  raise MissingAttributeError, "Missing attribute: '#{name}'." if attribute_missing?(name)
  attributes[name][:nillable]
end

def create_accessors(name)
  generated_methods.module_eval do
    define_method("#{name}=") do |value|
      write_attribute(name, value)
    end
    define_method(name) do
      read_attribute(name)
    end
    define_method("#{name}_before_typecast") do
      read_attribute_before_typecast(name)
    end
    define_method("#{name}?") do
      read_attribute(name).present?
    end
  end
end

def generated_methods

  end
end

def inspect
  attribute_types = attributes.map { |a, _| "#{a.to_s.colorize(:magenta)}: #{attribute_type(a)}" }
  class_name = name || sprintf('Class:%#016x', object_id << 1)
  "#{class_name}(#{attribute_types.join(', ')})"
end
