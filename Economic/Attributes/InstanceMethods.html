def ==(other)
  if self.equal?(other)
    true
  elsif other.instance_of?(self.class)
    keys = self.class.attributes.keys
    self.attributes.values_at(*keys) == other.attributes.values_at(*keys)
  else
    false
  end
end

def inspect
  attributes = self.class.attributes.map { |a, _| "#{a}:#{send(a).inspect}" }
  class_name = self.class.name || sprintf('Class:%#016x', self.class.object_id << 1)
  "#<#{class_name} #{attributes.join(' ')}>"
end

def to_api_hash
  hash = {}
  self.class.attributes.each do |name, options|
    converter = Economic.Converter(options[:type])
    value = read_attribute(name)
    if attribute_nillable?(name) || !value.nil?
      hash[name.to_s.camelcase] = converter.to_api(value)
    end
  end
  hash
end

def write_attribute(name, value)
  raise MissingAttributeError, "Missing attribute: '#{name}'." if attribute_missing?(name)
  attributes_before_typecast[name] = value
  typed_value = typed_value_for(name, value)
  attributes[name] = typed_value
end
