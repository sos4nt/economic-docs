def arguments(element)
  name = element[:name].underscore
  type = element[:type].sub(/^(s:|tns:)/, '').underscore
  array = true if type.sub!(/^array_of_/, '')
  nillable = element[:nillable]
  ary = []
  ary << ":#{name}"
  ary << "type: :#{type}"
  ary << "nillable: true" if nillable
  ary << "array: true" if array
  ary.join(', ')
end
