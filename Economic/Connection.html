#initialize(connection_method, credentials) ⇒ Connection

Returns a new connection object and connects using the specified connection_method and credentials.

You can connect either via token (recommended):

Economic::Connection.new(:connect_with_token, token: '...', app_token: '...')

or via agreement number, user name and password:

Economic::Connection.new(:connect, agreement_number: '...', user_name: '...', password: '...')

def self.open(connection_method, credentials)
  connection = new(connection_method, credentials)
  return connection unless block_given?
  begin
    yield connection
    nil
  ensure
    connection.close
  end
end

def call(method, message = nil)
  raise "Undefined operation '#{method}'" unless Economic.savon_client.operations.include?(method)
  if Economic.configuration.verbose
    t = Time.now
    printf "  %s %s\n", method.to_s.colorize(:cyan), Economic.help_url(method).light_green
    printf "  %s %s\n", '->'.colorize(:cyan), message.inspect
  end
  response = call_with_defaults(method, message: message)
  result = response.to_hash[:#{method}_response"][:#{method}_result"] if response
  yield response, result if block_given?
  result
ensure
  if Economic.configuration.verbose
    printf "  %s (%ims)\n", method.to_s.colorize(:magenta), (Time.now - t) * 100
    printf "  %s %s\n", '<-'.colorize(:magenta), result.inspect if response
  end
end

def call_with_defaults(method, hash={})
  hash = savon_defaults.merge(hash)
  Economic.savon_client.call(method, hash)
rescue Savon::SOAPFault => soap_error
  raise *economic_error(soap_error)
end

def economic_error(soap_error)
  match_data = soap_error.to_hash[:fault][:faultstring].match(/^(?<name>[\w.]+)\((?<code>\w+)\): (?<message>.+)$/) || {}
  code = match_data[:code]
  name = match_data[:name]
  message = "#{match_data[:message]} (#{code})"
  klass = Error.by_code(code) || Error.by_name(name)
  klass.new(soap_error, code, message) if klass
end
