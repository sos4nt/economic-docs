def connect(connection_method, credentials, &block)
  if block_given?
    Economic::Connection.open(connection_method, credentials) do |new_connection|
      with_connection(new_connection, &block)
    end
  else
    self.connection = Economic::Connection.open(connection_method, credentials)
  end
end

def with_connection(new_connection)
  temp = connection
  self.connection = new_connection
  yield connection
ensure
  self.connection = temp
end
