.fetch_array(result, key, default = []) ⇒ Object

Returns the key's value as an array.

result = { foo: {id: 1} }
ApiMethod.fetch_array(result, :foo) #=> [{id: 1}]

result = { foo: [{id: 1}, {id: 2}] }
ApiMethod.fetch_array(result, :foo) #=> [{id: 1}, {id: 2}]

result = { foo: nil }
ApiMethod.fetch_array(result, :foo) #=> [nil]

The default value is returned if result is nil or the key doesn't exist.

result = {}
ApiMethod.fetch_array(result, :foo) #=> []

result = {}
ApiMethod.fetch_array(result, :foo, nil) #=> nil

def self.fetch_array(result, key, default = [])
  return default unless result && result.has_key?(key)
  handles = result[key]
  handles = [handles] unless handles.is_a?(Array)
  handles
end

def options_hash
  hash = {}
  attributes.each do |attribute|
    converter = Economic.Converter(attribute.type)
    value = attribute.value
    if attribute.array?
      raise TypeError, "expected Array, got #{value.class}" unless value.is_a? Array
      values = value.map { |v| converter.to_api(converter.from_api(v)) }
      hash[attribute.name] = { attribute.type.to_s.camelcase => values }
    elsif attribute.nillable? || !value.nil?
      hash[attribute.name] = converter.to_api(converter.from_api(value))
    end
  end
  hash
end

def result_object
  return unless return_arg.type?
  converter = Economic.Converter(return_arg.type)
  if return_arg.array?
    ApiMethod.fetch_array(result, return_arg.type).map { |h| converter.from_api(h) }
  else
    converter.from_api(result)
  end
end
