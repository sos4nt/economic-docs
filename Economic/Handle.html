def get(attribute, type = nil)
  result = self.class.connection.call(:#{self.class.base_name}_get_#{attribute}", :#{self.class.base_name}_handle" => to_api_hash)
  if type
    Economic.converter_for_type(type).from_api(result)
  else
    result
  end
end

def set(attribute, value, type = nil)
  type ||= :#{self.class.base_name}_handle" if value.is_a? Handle
  if type
    converter = Economic.converter_for_type(type)
    value = converter.from_api(value)
    value = converter.to_api(value)
  end

  hash = { :#{self.class.base_name}_handle" => to_api_hash }
  if type.to_s.ends_with? '_handle'
    hash[:value_handle] = value
  else
    hash[:value] = value
  end
  self.class.connection.call(:#{self.class.base_name}_set_#{attribute}", hash)
end
